#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int i,j;
    
    for(i=0; i<board.size(); i++){
        for(j=0; j<board[i].size(); j++){
            if(i+j <= k){
                answer += board[i][j];
            }
        }
    }   
    
    return answer;
}
