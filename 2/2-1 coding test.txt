#include <string>
#include <vector>

using namespace std;

#define lamb    12000
#define drink   2000
    
int solution(int n, int k) {
    int answer = 0;
    
    for (int i=1; i<=n; i++){
        if(i%10 == 0 && k > 0){
            k -= 1;
        }   
    }
    
    n *= lamb;
    k *= drink;
    answer = (n+k);
    
    return answer;
}
